948 int VertexLaplacianVector(const mesh& meshin, const vert* vertsmooth,
949  coordvec &lapVec, bool isOrdered){
950 
951  coordvec tempPos;
952  coordvec temp1, temp2;
953  double totalCotan=0.0;
954 
955  // Make sure that we don't allocate or copy data if the laplacian is already
956  // ordered.
957  std::vector<int> edgeIndModif;
958  const std::vector<int>* edgeIndOutPtr = &(vertsmooth->edgeind);
959  if(!isOrdered){
960  auto retOrder = vertsmooth->OrderEdges(&meshin, edgeIndModif);
961  if (!rsvs3d::constants::ordering::__isordered(retOrder)){
962  return rsvs3d::constants::__failure;
963  }
964  edgeIndOutPtr = &edgeIndModif;
965  }
966  // Is a reference to either this->edgeind if isOrdered=true or edgeIndModif
967  // otherwise.
968  auto& edgindOrdered = *edgeIndOutPtr;
969 
970 
971  lapVec.assign(0.0,0.0,0.0);
972  // Cotangent multiplier cannot be defined (could fall back on uniform)
973  int nVerts = edgindOrdered.size();
974  if(nVerts<3){
975  return rsvs3d::constants::__failure;
976  }
977 
978  // Iterates around each
979  for (int i=0; i< nVerts; ++i){
980 
981  int connVert = meshin.VertFromVertEdge(vertsmooth->index,
982  edgindOrdered[i]);
983  int connVert_m1 = meshin.VertFromVertEdge(vertsmooth->index,
984  edgindOrdered[(i-1+nVerts)%nVerts]);
985  int connVert_p1 = meshin.VertFromVertEdge(vertsmooth->index,
986  edgindOrdered[(i+1)%nVerts]);
987  tempPos = meshin.verts.isearch(connVert)->coord;
988 
989  double currCotan = CotanLaplacianWeight(vertsmooth->coord,
990  meshin.verts.isearch(connVert_m1)->coord,
991  meshin.verts.isearch(connVert)->coord,
992  meshin.verts.isearch(connVert_p1)->coord,
993  temp1,temp2);
994 
995  if(!isfinite(currCotan)){
996  return rsvs3d::constants::__failure;
997  }
998  tempPos.mult(currCotan);
999  totalCotan += currCotan;
1000  lapVec.add(tempPos.usedata());
1001 
1002  }
1003 
1004  lapVec.div(totalCotan);
1005  lapVec.substract(vertsmooth->coord);
