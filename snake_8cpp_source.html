437 int CompareSnakeInternalStatus(const vector<bool> & thisVec,bool thisFlipped,
438  const vector<bool> & otherVec, bool otherFlipped){
439  /*
440  Checks if a snake is inside another using information from the points inside
441  the snaking mesh.
442  Works by finding counter examples.
443  This is not a complete test and assumes a relatively simple relationship between
444  the two snakes. of either one completely internal or the two side by side.
445  */
446  bool isIndep;
447  int ii, ni;
448  ni=thisVec.size();
449  isIndep=true;
450  for (ii=0; ii<ni; ++ii){
451  if((thisVec[ii]^thisFlipped) && (otherVec[ii]^otherFlipped)){
452  isIndep = false;
453  break;
454  }
455  }
456 
457  return(int(isIndep));
458 }
