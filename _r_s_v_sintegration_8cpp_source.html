369 void integrate::prepare::grid::Voxel(
370  const param::grid &gridconf,
371  mesh &snakeMesh,
372  mesh &voluMesh
373  ){
374 
375  int ii;
376  auto gridSize = gridconf.voxel.gridsizebackground;
377  vector<int> elmMapping, backgroundGrid;
378 
379  backgroundGrid.reserve(int(gridSize.size()));
380  for (int i = 0; i < int(gridSize.size()); ++i)
381  {
382  backgroundGrid.push_back(gridSize[i]);
383  gridSize[i] = gridSize[i] * gridconf.voxel.gridsizesnake[i];
384  }
385 
386  // Initial build of the grid
387  BuildBlockGrid(gridSize, snakeMesh);
388  snakeMesh.Scale(gridconf.domain);
389  snakeMesh.PrepareForUse();
390  snakeMesh.OrientFaces();
391 
392  // map elements to coarse grid
393  if(snakeMesh.WhatDim()==3){
394  for (ii=0;ii<snakeMesh.volus.size();++ii){
395  elmMapping.push_back(1);
396  }
397  } else if (snakeMesh.WhatDim()==2){
398  for (ii=0;ii<snakeMesh.surfs.size();++ii){
399  elmMapping.push_back(1);
400  }
401  } else {
402  RSVS3D_ERROR_ARGUMENT("Incorrect dimension");
403  }
404  CartesianMapping(snakeMesh, elmMapping, backgroundGrid);
405  CoarsenMesh(snakeMesh,voluMesh,elmMapping);
406  snakeMesh.AddParent(&voluMesh,elmMapping);
