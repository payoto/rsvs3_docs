347

352  snakeMesh.OrientFaces();
353 
354  // map elements to coarse grid
355  if(snakeMesh.WhatDim()==3){
356  for (ii=0;ii<snakeMesh.volus.size();++ii){
357  elmMapping.push_back(1);
358  }
359  } else if (snakeMesh.WhatDim()==2){
360  for (ii=0;ii<snakeMesh.surfs.size();++ii){
361  elmMapping.push_back(1);
362  }
363  } else {
364  RSVS3D_ERROR_ARGUMENT("Incorrect dimension");
365  }
366  CartesianMapping(snakeMesh, elmMapping, backgroundGrid);
367  CoarsenMesh(snakeMesh,voluMesh,elmMapping);
