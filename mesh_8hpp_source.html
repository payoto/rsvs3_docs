789 double meshhelp::ProjectRay(int count, const W &&boundBox,
790  const T &dir, const V &orig, double minDist){
791  /*
792  Calculates the distance to project a single ray to reach the bounding box
793 
794  It is templated to accept any types of containers with `count` elements,
795  accessible by operator[];
796 
797  W is an object of size [2][count] accessible by operators [][]
798  */
799 
800  double l=-INFINITY;
801 
802  for (int i = 0; i < count; ++i)
803  {
804  l = std::max(l,
805  std::min(
806  (boundBox[0][i]-orig[i])/dir[i] ,
807  (boundBox[1][i]-orig[i])/dir[i] ));
808  }
809  return std::min(l, minDist);
810 }
