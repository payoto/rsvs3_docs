252 double tetgen::internal::ProjectRay(int count, const tetgen::dombounds &boundBox,
253  const T &dir, const V &orig, double minDist){
254  /*
255  Calculates the distance to project a single ray to reach the bounding box
256 
257  It is templated to accept any types of containers with `count` elements,
258  accessible by operator[];
259  */
260 
261  double l=-INFINITY;
262 
263  for (int i = 0; i < count; ++i)
264  {
265  l = std::max(l,
266  std::min(
267  (boundBox[0][i]-orig[i])/dir[i] ,
268  (boundBox[1][i]-orig[i])/dir[i] ));
269  }
270  return std::min(l, minDist);
271 }
